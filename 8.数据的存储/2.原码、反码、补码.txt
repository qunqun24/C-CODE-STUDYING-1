#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h>

int main()
{
	//int-4个字节-32个比特位
	int a = 20;//内存：14 00 00 00
	//原码（为正，第一位符号位为0）：                  00000000 00000000 00000000 00010100
	//正数原码、反码、补码相同
	//转化成16进制（4个2进制位转为为一个16进制位）：0x00       00       00       14

	int b = -10;//内存：f6 ff ff ff（倒着存放的，后面有讲大端和小端存储模式，这里是小端存储）
	//原码（为负，第一位符号位为1）：               10000000 00000000 00000000 00001010
	//反码（原码的符号位不变，其他位依次按位取反）：11111111 11111111 11111111 11110101
	//补码（反码+1）：                              11111111 11111111 11111111 11110110
	//补码换算成2进制位：                         0xff       ff       ff       f6
	//结论：对于整形来说，数据存放内存中其实存放的是补码（倒着存放的）


	1 - 1;
	//1+(-1)
	//1(原码)：     00000000 00000000 00000000 00000001
	//-1（原码）：  10000000 00000000 00000000 00000001
	//若是原码相加：10000000 00000000 00000000 00000010 （是-2，不正确）

	//1（补码）：   00000000 00000000 00000000 00000001
	//-1（补码）：  11111111 11111111 11111111 11111111
	//若是补码相加：00000000 00000000 00000000 00000000 （是0，正确）
	//结论：使用补码，加法和减法也可以统一处理（CPU只有加法器）
	return 0;
}
