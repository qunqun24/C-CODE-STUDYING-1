#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h>
#include<windows.h>//Sleep头文件

int main()
{
	unsigned int i;
	for (i = 9; i >= 0; i--)
	{
		printf("%u\n", i);
		Sleep(100);//睡眠100ms=0.1s
	}
}
/*
输出：
9
8
7
6
5
4
3
2
1
0
4294967295
4294967294
4294967293
4294967292
......(循环无限执行)
*/

//i=9
//i:00000000 00000000 00000000 00001001
//输出无符号十进制即9
//......
//i=0-1;因为i是无符号数，一定大于0，所以循环无限执行
//-1原码：10000000 00000000 00000000 00000001
//-1补码：11111111 11111111 11111111 11111111（即i）
//输出无符号数：4,294,967,295
//......
