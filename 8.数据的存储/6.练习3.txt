#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h>

int main()
{
	int i = -20;
	//-20原码：10000000 00000000 00000000 00010100
	//-20补码：11111111 11111111 11111111 11101100（即i）

	unsigned int j = 10;
	//10原码： 00000000 00000000 00000000 00001010（原反补相同，即j）

	printf("%d\n", i + j);
	//i+j:     11111111 11111111 11111111 11110110（补码运算）
	//求原码： 10000000 00000000 00000000 00001010（即-10）
	           
	return 0;
	//按照补码的形式进行运算，最后格式化成为有符号整数
}
