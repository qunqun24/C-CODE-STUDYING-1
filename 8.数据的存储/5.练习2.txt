#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h>

int main()
{
	char a = -128;//（有符号）
	//-128原码：10000000 00000000 00000000 10000000
	//-128补码：11111111 11111111 11111111 10000000
	//a中存放：                            10000000（char a;按符号位1往前补1）
	//整型提升：11111111 11111111 11111111 10000000（补码，要打印无符号数字，原码、反码、补码相同）化成十进制是4,294,967,168

	printf("%u\n", a);
	//%d - 打印十进制的有符号数字
	//%u - 打印十进制的无符号数字（这里的第一位1就不是符号位了，所以输出出来数字会很大）
	//输出：4,294,967,168

	printf("%d\n", a);
	//整型提升：11111111 11111111 11111111 10000000（补码)
	//原码：    10000000 00000000 00000000 10000000 （输出有符号数）
	//输出：-128
	return 0;
}
