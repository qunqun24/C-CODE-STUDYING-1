#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h>

int main()
{
	char a = -1;//（有符号）
	//-1的原码：10000000 00000000 00000000 00000001
	//-1的补码：11111111 11111111 11111111 11111111
	//char a,a中存放：                     11111111（认为最高位1就是符号位，前面都补符号位1）
	//a打印%d： 11111111 11111111 11111111 11111111 -> 然后倒回去算原码（反码=补码-1；原码等于补码符号位不变，其他位按位取反）
	//打印a=-1

	signed char b = -1;//（有符号）
	//signed char b，b中存放：             11111111
	//与a一样
	//打印b=-1

	unsigned char c = -1;//（无符号）
	//unsigned char c，c中存放：           11111111（无符号数高位补0）
	//c打印%d： 00000000 00000000 00000000 11111111（符号位是0，正数原码、反码、补码相同）
	//打印c=255

	printf("a=%d,b=%d,c=%d", a, b, c);//整形提升
	//输出：a=-1,b=-1,c=255
	return 0;
}
