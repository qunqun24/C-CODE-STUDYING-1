#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h>

int main()
{
	int a = 16;
	//>>    右移操作符
	//移动的是二进制位
	//00000000 00000000 00000000 00010000
	int b = a >> 1;
	//000000000 00000000 00000000 0001000
	printf("16=a >> 1 = %d\n", b);

	//以下判断是算术右移（负数仍为负数）还是逻辑右移（负数变正数）
	printf("\n");
	int a1 = -1;
	int b1 = a1 >> 1;
	printf("-1=a1 >> 1 = %d\n", b1);
	//算术右移

	//a1=-1
	//整数的二进制表示有:原码、反码、补码
	//存储到内存的是补码
	//所以移位移的也是补码
        //10000000000000000000000000000001 - 原码(有符号位)
	//11111111111111111111111111111110 - 反码(原码符号位不变 其余位按位取反)
	//11111111111111111111111111111111 - 补码(反码加一) 算术移位后不变仍为-1
}
